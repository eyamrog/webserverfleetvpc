#!/bin/bash
yum -y update
yum install -y httpd
aws s3 cp s3://eyamrog-bucket/index.html /var/www/html/
aws s3 cp s3://eyamrog-bucket/elbhealthcheck.html /var/www/html/

chown -R apache.apache /var/www/html/
chmod -R 755 /var/www/html/
chkconfig httpd on
